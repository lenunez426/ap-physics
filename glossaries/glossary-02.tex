\makenoidxglossaries

\newglossaryentry{position}{
    name=position,
    description={the location of an object at a particular time}
}

\newglossaryentry{displacement}{
    name=displacement,
    description={the change in position of an object}
}

\newglossaryentry{distance}{
    name=distance,
    description={the magnitude of displacement between two positions}
}

\newglossaryentry{distance traveled}{
    name=distance traveled,
    description={the total length of the path traveled between two positions}
}

\newglossaryentry{vector}{
    name=vector,
    description={a quantity that is described by both magnitude and direction}
}

\newglossaryentry{scalar}{
    name=scalar,
    description={a quantity that is described by magnitude, but not direction}
}
